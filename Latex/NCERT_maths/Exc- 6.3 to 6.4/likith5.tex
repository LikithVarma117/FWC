\end{center}

\begin{enumerate}[label=\textbf{\color{blue}\arabic*.}]

\item Find the slope of the tangent to the curve
\begin{align*}
y = 3x^4 - 4x
\end{align*}
at $x = 4$.

\item Find the slope of the tangent to the curve
\begin{align*}
y = \dfrac{x-1}{x-2}, x \neq 2
\end{align*}
at $x = 10$.

\item Find the slope of the tangent to curve
\begin{align*}
y = x^3 - x + 1
\end{align*}
at the point whose
$x$-coordinate is $2$.

\item Find the slope of the tangent to the curve
\begin{align*}
y = x^3 - 3x + 2
\end{align*}
at the point whose
$x$-coordinate is $3$.

\item Find the slope of the normal to the curve
\begin{align*}
x & = a\cos^3\theta\\
y & = a\sin^3\theta
\end{align*}
at $\theta = \dfrac{\pi}{4}$.

\item Find the slope of the normal to the curve
\begin{align*}
x &  = 1 - a\sin\theta\\
y &  = b\cos^2\theta
\end{align*}
at $\theta = \dfrac{\pi}{2}$.

\item Find points at which the tangent to the curve
\begin{align*}
y = x^3 - 3x^2 - 9x + 7
\end{align*}
is parallel to the
$x$-axis.

\item Find a point on the curve
\begin{align*}
y = (x - 2)^2
\end{align*}
at which the tangent is
parallel to the chord joining the points
$(2, 0)$ and $(4, 4)$.

\item Find the point on the curve
\begin{align*}
y = x^3 - 11x + 5
\end{align*}
at which the tangent is
$y = x - 11$.

\item Find the equation of all lines having slope $-1$
that are tangents to the curve
\begin{align*}
y = \dfrac{1}{x-1}, x \neq 1.
\end{align*}

\item Find the equation of all lines having slope $2$
which are tangents to the curve
\begin{align*}
y = \dfrac{1}{x-3}, x \neq 3.
\end{align*}

\item Find the equations of all lines having slope $0$
which are tangent to the curve
\begin{align*}
y = \dfrac{1}{x^2 - 2x + 3}.
\end{align*}

\item Find points on the curve
$\dfrac{x^2}{9} + \dfrac{y^2}{16} = 1$ at which the tangents are
\begin{enumerate}[label=(\roman*)]
\item parallel to $x$-axis
\item parallel to $y$-axis.
\end{enumerate}

\item Find the equations of the tangent and normal to the
given curves at the indicated points:
\begin{enumerate}[label=(\roman*)]
\item $y = x^4 - 6x^3 + 13x^2 - 10x + 5$ at $(0, 5)$
\item $y = x^4 - 6x^3 + 13x^2 - 10x + 5$ at $(1, 3)$
\item $y = x^3$ at $(1, 1)$
\item $y = x^2$ at $(0, 0)$
\item $x = \cos t$, $y = \sin t$ at $t = \dfrac{\pi}{4}$
\end{enumerate}

\item Find the equation of the tangent line to the curve
$y = x^2 - 2x + 7$ which is
\begin{enumerate}[label=(\alph*)]
\item parallel to the line $2x - y + 9 = 0$
\item perpendicular to the line $5y - 15x = 13$.
\end{enumerate}

\item Show that the tangents to the curve
\begin{align*}
y = 7x^3 + 11
\end{align*}
at the points where $x = 2$ and
$x = -2$ are parallel.

\item Find the points on the curve
\begin{align*}
y = x^3
\end{align*}
at which the slope of the tangent is
equal to the $y$-coordinate of the point.

\item For the curve
\begin{align*}
y = 4x^3 - 2x^5,
\end{align*}
find all the points at which the
tangent passes through the origin.

\item Find the points on the curve
\begin{align*}
x^2 + y^2 - 2x - 3 = 0
\end{align*}
at which the tangents are
parallel to the $x$-axis.

\item Find the equation of the normal at the point
$(am^2, am^3)$ for the curve
\begin{align*}
ay^2 = x^3.
\end{align*}

\item Find the equation of the normals to the curve
\begin{align*}
y = x^3 + 2x + 6
\end{align*} which are parallel to the line
$x + 14y + 4 = 0$.

\item
Find the equations of the tangent and normal to the
parabola 
\begin{align*}
y^2 = 4ax
\end{align*}
at the point $(at^2, 2at)$.

\item
Prove that the curves 
\begin{align*}
x = y^2, \quad xy = k 
\end{align*}
cut at right angles if $8k^2 = 1$.

\item
Find the equations of the tangent and normal to the
hyperbola
\begin{align*}
\dfrac{x^2}{a^2} - \dfrac{y^2}{b^2} = 1
\end{align*}
at the point $(x_0, y_0)$.

\item Find the equation of the tangent to the curve
\begin{align*}
y = \sqrt{3x - 2}
\end{align*}
which is parallel to the line
$4x - 2y + 5 = 0$.

\end{enumerate}
